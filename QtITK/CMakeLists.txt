PROJECT(QtITKExample)

# 
# Find ITK
#
INCLUDE (${CMAKE_ROOT}/Modules/FindITK.cmake)
IF (USE_ITK_FILE)
  INCLUDE(${USE_ITK_FILE})
ENDIF(USE_ITK_FILE)

# 
# Find QT
#
INCLUDE (${CMAKE_ROOT}/Modules/FindQt.cmake)

INCLUDE_DIRECTORIES(
${QT_INCLUDE_DIR}
)

LINK_DIRECTORIES(
${ITK_BINARY_DIR}/Code/Common
${ITK_BINARY_DIR}/Code/Numerics
${ITK_BINARY_DIR}/Code/Numerics/vxl
${ITK_BINARY_DIR}/Code/BasicFilters
${ITK_BINARY_DIR}/Code/Algorithms
${ITK_BINARY_DIR}/Code/IO
${ITK_BINARY_DIR}/Utilities
${ITK_BINARY_DIR}/Utilities/png
${ITK_BINARY_DIR}/Utilities/zlib
${ITK_BINARY_DIR}/Utilities/MetaIO
)

LINK_LIBRARIES (
ITKIO
${QT_QT_LIBRARY}
)

SET(QtITK_SRCS
itkQtProgressBar.cxx
qtITK.cxx
)

SET(QtITK_MOC_SRCS 
itkQtAdaptor.h
itkQtLightIndicator.h
)

IF(QT_WRAP_CPP)
  QT_WRAP_CPP(QtITK QtITK_SRCS QtITK_MOC_SRCS)
ENDIF(QT_WRAP_CPP)


ADD_EXECUTABLE(QtITK QtITK_SRCS)
